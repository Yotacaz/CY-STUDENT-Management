72static inline float get_student_general_avg(Student *stu, CoursesTab *ctab)
73{
74 float total_grade = 0;
75 float total_coef = 0;
76 for (int i = 0; i < ctab->size; i++)
77 {
78 float avg = stu->f_courses[i]->average;
79 if (avg > GRADE_MIN && avg < GRADE_MAX)
80 {
81 float coef = ctab->tab[i]->coef;
82 total_grade += stu->f_courses[i]->average * coef;
83 total_coef += coef;
84 }
85 }
86 return total_coef > 0 ? total_grade / total_coef : -1;
87}
